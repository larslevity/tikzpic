\def\a{1.2}   % width of ellipse
\def\b{.6}	% height of ellipse
\def\c{.5}	% height of disk

\newcommand{\element}[4]{
	\def\xmid{#1}
	\def\ymid{#2}
	\def\col{#3}
	\def\ttext{#4}

	\draw[fill=\col] (\xmid, \ymid-\c) ellipse [x radius=\a,y radius=\b];
	\fill[\col] (\xmid-\a,\ymid)rectangle(\xmid+\a,\ymid-\c);
	\draw (\xmid-\a,\ymid)--++(0,-\c) (\xmid+\a,\ymid)--++(0,-\c);
	\draw[fill=gray!10] (\xmid, \ymid) ellipse [x radius=\a,y radius=\b];
	\path (\xmid, \ymid) node[align=center]{\ttext};
}


\element{0}{2}{blue}{\textbf{Mensch} \\ Verstand $\cdot$ Emotionen \\ Werte};

\newcommand{\arrows}[3]{  % center1, center2, shiftfactor
	\pgfmathsetmacro{\posi}{.4 - #3};
	\pgfmathsetmacro{\posii}{.6 - #3};
	\path (#1)--(#2)coordinate[pos=\posi](A) coordinate[pos=\posii](B);
	\draw[latex-latex, very thick] (A)--(B);
	}

\arrows{0,2}{-2,0}{-.08};
\arrows{0,2}{2,0}{-.08};
\arrows{-2,0}{2,0}{0};
