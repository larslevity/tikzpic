\documentclass[12pt]{standalone}
\input{../../tikzpic_packages.tex}

\usepackage{setspace}
\setstretch{.75}	%% linespacing 



\begin{document}
\begin{tikzpicture}[
	level0/.style={rounded corners=5mm,thick,fill=blue!15},
	pooldescript/.style={rotate=90,below,align=center, black},
	level1/.style={rounded corners=3mm,thick,fill=blue!23,minimum width=4.6cm,align=center},
	level2/.style={rounded corners=2mm,thick,fill=blue!50, text=white,minimum width=3cm},
	thickArrow/.style={thick, -stealth},
%
	matrix1/.style={matrix of nodes,row sep=2.5mm,level1},
	title/.style={draw=none,inner sep=0pt}
]
\newcommand{\matrixnode}[2]{
	\path (0,0)node[level2, align=center](#1){#1 \\ \tiny #2};
}
\newcommand{\othernode}[3]{
	\path #1 node[level2, align=center](#2){#2 \\ \tiny #3};
}



%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\matrix[matrix1] (concept) at (0,0){
    |[title]| \textsc{Conceptual Design} \\
    \matrixnode{type}{long long long type description} \\
    \matrixnode{number}{number description} \\
    \matrixnode{arrangement}{(serial, parallel, hybrid)} \\
    \matrixnode{connection}{connection description} \\
};
\path (concept.0)node[pooldescript](elementsDescription){
	elements \\ \tiny (e.\,g., actuators, suction cups, etc.)};
\begin{pgfonlayer}{background}
  \node[level0] [fit = (concept) (elementsDescription)] {};
\end{pgfonlayer}


\draw[thickArrow] (type)--(number);
\draw[thickArrow] (number)--(arrangement);
\draw[thickArrow] (arrangement)--(connection);
\draw[thickArrow] (connection.5)--++(.2,0)|-(arrangement.-5);
\draw[thickArrow] (arrangement.5)--++(.2,0)|-(number.-5);
%\draw[thickArrow] (number.5)--++(.2,0)|-(type.-5);   % standard
\draw[thickArrow] (number.5)--++(.6,0)|-(type.-5);		% for long description


%\draw[thickArrow] (connection.5)--++(.2,0)|-(type.-5);

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\othernode{($(concept.90)+(0,1)$)}{solution}{solution description}
\othernode{($(solution.90)+(0,1)$)}{task}{task description}
\othernode{($(concept.270)+(0,-1)$)}{model}{model description}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%


\matrix[matrix1, below] (geometry) at ($(model.270)+(0,-.5)$){
    |[title]| \textsc{Geometrical Design} \\
    \matrixnode{shape}{shape description} \\
    \matrixnode{material}{material description} \\
    \matrixnode{surface}{surface description} \\
    \matrixnode{dimension}{dimension description} \\
};
\path (geometry.0)node[pooldescript](fabricationDescription){
	fabrication \\ \tiny (fabricationDescription)};
\begin{pgfonlayer}{background}
  \node[level0] [fit = (geometry) (fabricationDescription)] {};
\end{pgfonlayer}


\draw[thickArrow] (shape)--(material);
\draw[thickArrow] (material)--(surface);
\draw[thickArrow] (surface)--(dimension);
\draw[thickArrow] (dimension.5)--++(.2,0)|-(surface.-5);
\draw[thickArrow] (surface.5)--++(.2,0)|-(material.-5);
\draw[thickArrow] (material.5)--++(.2,0)|-(shape.-5);

%\draw[thickArrow] (dimension.5)--++(.2,0)|-(shape.-5);

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\othernode{($(geometry.270)+(0,-1)$)}{realization}{realization description}

\draw[thickArrow] (task) -- (solution);
\draw[thickArrow] (solution) -- (concept);
\draw[thickArrow] (concept) -- (model);
\draw[thickArrow] (model) -- (geometry);
\draw[thickArrow] (geometry) -- (realization);

\draw[thickArrow] (model.east)-|($(concept.south)+(2,0)$);
\draw[thickArrow] (realization.east)-|($(geometry.south)+(2,0)$);


\end{tikzpicture}
